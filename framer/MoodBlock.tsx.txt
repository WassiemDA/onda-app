import React from "react"

type Track = {
  title: string
  artist: string
  explanation: string
  spotifyId: string
}

type Playlist = {
  title: string
  description: string
  tracks: Track[]
}

type Props = {
  persona: {
    playlists?: Playlist[]
  }
}

export function MoodBlock({ persona }: Props) {
  if (!persona?.playlists || !Array.isArray(persona.playlists)) {
    console.warn("⚠️ MoodBlock: No valid playlists found in persona:", persona)
    return <p>🕵️ No mood-based playlists available.</p>
  }

  return (
    <div style={{ marginTop: "2rem" }}>
      {persona.playlists.map((playlist, index) => (
        <div key={index} style={{ marginBottom: "2rem" }}>
          <h3 style={{ marginBottom: "0.5rem" }}>{playlist.title}</h3>
          <p style={{ fontStyle: "italic", marginBottom: "1rem" }}>
            {playlist.description}
          </p>
          <ul>
            {playlist.tracks.map((track, idx) => (
              <li key={idx} style={{ marginBottom: "1rem" }}>
                <strong>{track.title}</strong> by {track.artist}
                <br />
                <small>{track.explanation}</small>
                <br />
                {track.spotifyId ? (
                  <a
                    href={`https://open.spotify.com/track/${track.spotifyId}`}
                    target="_blank"
                    rel="noreferrer"
                  >
                    🔗 Listen on Spotify
                  </a>
                ) : (
                  <span>🔍 Preview unavailable (no Spotify link)</span>
                )}
              </li>
            ))}
          </ul>
        </div>
      ))}
    </div>
  )
}
